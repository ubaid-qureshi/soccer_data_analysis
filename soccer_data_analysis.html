def pick_year(x):
    #y = datetime.strptime(x, '%Y %m %d %I:%M%:%S' )
    z = str(x).split('-')[0]
    return int(z)

def arranging_data(s,l,limit = 10):
    for x in l:
        if x not in s:
            s.append(x)
            if len(s) == limit:
                return s
    return s
